@startuml
title Algorithme d'orientation COVID19

start

if (SI moins de 15 ans) then
   :FIN1;
   stop

elseif (SI 1 OU plus facteur de gravité majeur) then (oui)
       :FIN5;
       stop

elseif (SI fièvre ET toux) then (oui)
       if (SI >=1 facteur pronostique et >= 2 facteurs de gravité mineurs) then
          :FIN4;
          stop
       else (non)
          :FIN6;
          stop
       endif

elseif (SI fièvre OU diarrhée OU (toux ET douleurs) OU (toux ET anosmie)) then (oui)
        if (SI >= 1 facteur pronostique) then
           if (SI >=2 facteurs de gravité mineurs) then (oui)
              :FIN4;
               stop
           else (non)
              :FIN3;
              stop
           endif
        else 
            if (SI >= 50 ans ou >=1 facteur pronostique) then (oui)
               :FIN3;
               stop
            else (non)
               :FIN2; 
               stop
            endif
        endif

elseif (SI pas de fièvre ET (toux OU douleurs OU anosmie)) then (oui)
       if (SI >=1 facteur de gravité mineur OU >=1 facteur pronostique) then (oui)
          :FIN8; 
          stop
       else (non)
          :FIN7;
          stop
       endif

else
       if (SI NI toux NI douleurs NI anosmie) then (oui)
         :FIN9;
         stop
       endif
endif

:FIN9;

@enduml
